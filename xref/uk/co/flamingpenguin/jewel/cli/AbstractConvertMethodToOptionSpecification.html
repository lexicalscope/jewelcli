24  class AbstractConvertMethodToOptionSpecification {
25      protected final ReflectedClass<?> klass;
26  
27      public AbstractConvertMethodToOptionSpecification(final ReflectedClass<?> klass) {
28          this.klass = klass;
29      }
30  
31      protected UnparsedOptionSpecification createUnparsedOptionSpecificationFrom(
32              final ReflectedMethod method) {
33          final UnparsedOptionSpecificationBuilder optionSpecificationBuilder =
34                  new UnparsedOptionSpecificationBuilder(method);
35  
36          final UnparsedAnnotationAdapter annotation =
37                  new UnparsedAnnotationAdapter(klass, method, method.annotation(Unparsed.class));
38  
39          configureSpecificationFromAnnotation(optionSpecificationBuilder, annotation);
40  
41          optionSpecificationBuilder.setValueName(annotation.name());
42  
43          return optionSpecificationBuilder.createOptionSpecification();
44      }
45  
46      protected ParsedOptionSpecification createParsedOptionSpecificationFrom(
47              final ReflectedMethod method) {
48          final ParsedOptionSpecificationBuilder optionSpecificationBuilder =
49                  new ParsedOptionSpecificationBuilder(method);
50  
51          final OptionAnnotationAdapter annotation = new OptionAnnotationAdapter(klass, method);
52  
53          optionSpecificationBuilder.setShortNames(annotation.shortName());
54          optionSpecificationBuilder.setLongName(annotation.longName());
55          optionSpecificationBuilder.setHelpRequest(annotation.helpRequest());
56  
57          configureSpecificationFromAnnotation(
58                  optionSpecificationBuilder,
59                  annotation);
60  
61          return optionSpecificationBuilder.createOptionSpecification();
62      }
63  
64      private void configureSpecificationFromAnnotation(
65              final OptionSpecificationBuilder optionSpecificationBuilder,
66              final OptionAdapter annotation) {
67  
68          optionSpecificationBuilder.setDescription(annotation.description().trim());
69          optionSpecificationBuilder.setPattern(annotation.pattern());
70          optionSpecificationBuilder.setOptionalityMethod(annotation.correspondingOptionalityMethod());
71          optionSpecificationBuilder.setType(annotation.getValueType());
72          optionSpecificationBuilder.setMultiValued(annotation.isMultiValued());
73  
74          if (annotation.defaultToNull() && annotation.hasDefaultValue())
75          {
76              throw new OptionSpecificationException("option cannot have null default and non-null default value: "
77                      + annotation.method());
78          }
79          else if (annotation.defaultToNull())
80          {
81              optionSpecificationBuilder.setDefaultToNull(true);
82              if (annotation.isMultiValued())
83              {
84                  optionSpecificationBuilder.setDefaultValue(null);
85              }
86              else
87              {
88                  optionSpecificationBuilder.setDefaultValue(asList((String) null));
89              }
90          }
91          else if (annotation.hasDefaultValue())
92          {
93              optionSpecificationBuilder.setDefaultValue(asList(annotation.defaultValue()));
94          }
95      }
