7   import java.util.List;

9   import com.lexicalscope.fluentreflection.ReflectedMethod;

11  /**
12   * Specifies an Option
13   * 
14   * @author t.wood
15   */
16  interface OptionSpecification
17  {
18      /**
19       * Each argument to this option must conform to the type returned by this
20       * method
21       * 
22       * @return the type that each argument must conform to
23       */
24      Class<?> getType();
25  
26      /**
27       * Canonical identifier for the option
28       * 
29       * @return the canonical identifier for the option
30       */
31      String getCanonicalIdentifier();
32  
33      /**
34       * Are multiple arguments allowed?
35       * 
36       * @return True iff the the option takes multiple arguments
37       */
38      boolean isMultiValued();
39  
40      /**
41       * Is the argument optional
42       * 
43       * @return is the argument optional
44       */
45      boolean isOptional();
46  
47      /**
48       * Get the default values which will be used if the option is not specified
49       * by the user.
50       * 
51       * @return The default values which will be used if the option is not
52       *         present
53       */
54      List<String> getDefaultValue();
55  
56      /**
57       * Is there a default value to use if this option is not present? Options
58       * with a default value are assumed to be optional.
59       * 
60       * @return True iff this option has a default value
61       */
62      boolean hasDefaultValue();
63  
64      ReflectedMethod getMethod();
65  
66      ReflectedMethod getOptionalityMethod();
67  }
