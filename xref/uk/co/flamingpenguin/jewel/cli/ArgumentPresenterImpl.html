25          final Map<String, Object> argumentMap = new LinkedHashMap<String, Object>();
26  
27          final ValidationErrorBuilder validationErrorBuilder = new ValidationErrorBuilderImpl();
28  
29          final List<ReflectedMethod> optionMethods = klass.methods(annotatedWith(Option.class));
30          for (final ReflectedMethod reflectedMethod : optionMethods) {
31              final OptionSpecification optionSpecification =
32                      specification.getSpecification(reflectedMethod);
33  
34              final ConvertTypeOfObject<?> convertTypeOfObject =
35                      converterTo(validationErrorBuilder, optionSpecification, reflectedMethod.returnType());
36              final Option optionAnnotation = reflectedMethod.annotation(Option.class);
37              if (reflectedMethod.returnType().assignableTo(Iterable.class))
38              {
39                  argumentMap.put(
40                          reflectedMethod.propertyName(),
41                          convertTypeOfObject.convert(asList(optionAnnotation.defaultValue())));
42              }
43              else if (optionAnnotation.defaultValue() != null && optionAnnotation.defaultValue().length > 0)
44              {
45                  argumentMap.put(
46                          reflectedMethod.propertyName(),
47                          convertTypeOfObject.convert(optionAnnotation.defaultValue()[0]));
48              }
49  
50              final Argument argument = validatedArguments.getArgument(optionSpecification.getNames());
51              if (argument != null) {
52                  if (optionSpecification.isMultiValued()) {
53                      argumentMap.put(
54                              optionSpecification.getLongName(),
55                              convertTypeOfObject.convert(argument.getValues()));
56                  } else if (!argument.getValues().isEmpty()) {
57                      argumentMap.put(
58                              optionSpecification.getLongName(),
59                              convertTypeOfObject.convert(argument.getValues().get(0)));
60                  } else if (argument.getValues().isEmpty()) {
61                      argumentMap.put(optionSpecification.getLongName(), null);
62                  }
63              }
64          }
65  
66          if (specification.hasUnparsedSpecification()) {
67              final List<ReflectedMethod> unparsedMethods = klass.methods(annotatedWith(Unparsed.class));
68              for (final ReflectedMethod reflectedMethod : unparsedMethods) {
69                  final ConvertTypeOfObject<?> convertTypeOfObject =
70                          converterTo(
71                                  validationErrorBuilder,
72                                  specification.getUnparsedSpecification(),
73                                  reflectedMethod.returnType());
74                  if (!validatedArguments.getUnparsed().isEmpty())
75                  {
76                      if (reflectedMethod.returnType().assignableTo(Iterable.class))
77                      {
78                          argumentMap.put(

80                                  convertTypeOfObject.convert(validatedArguments.getUnparsed()));
81                      }
82                      else if (!validatedArguments.getUnparsed().isEmpty())
83                      {

86                                  convertTypeOfObject.convert(validatedArguments.getUnparsed().get(0)));
87                      }
88                  }
89              }
90          }
91          validationErrorBuilder.validate();
