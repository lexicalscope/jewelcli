165     public ArgumentValidationException(final List<ValidationError> validationErrors)
166     {
167         m_validationErrors = new ArrayList<ValidationError>(validationErrors);
168 
169         final StringBuilder message = new StringBuilder();
170 
171         String separator = "";
172         for (final ValidationError error : validationErrors)
173         {
174             message.append(separator).append(error.toString());
175             separator = System.getProperty("line.separator");
176         }
177         m_message = message.toString();
178     }
