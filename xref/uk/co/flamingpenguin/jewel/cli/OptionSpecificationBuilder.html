14  package uk.co.flamingpenguin.jewel.cli;
15  
16  import java.lang.reflect.Method;
17  import java.util.ArrayList;
18  import java.util.List;
19  
20  import com.lexicalscope.fluentreflection.ReflectedMethod;
21  
22  class OptionSpecificationBuilder {
23      private final Method m_method;
24      private Class<?> m_type;
25      private boolean m_multiValued;
26      private ReflectedMethod optionalityMethod;
27      private final List<String> m_shortNames = new ArrayList<String>();
28      private String m_longName;
29      private String m_description;
30      private String m_pattern;
31      private List<String> m_defaultValue;
32      private boolean m_helpRequest;
33  
34      public OptionSpecificationBuilder(final Method method) {
35          m_method = method;
36      }
37  
38      public void setType(final Class<?> type) {
39          m_type = type;
40      }
41  
42      public void setMultiValued(final boolean multiValued) {
43          m_multiValued = multiValued;
44      }
45  
46      public void setOptionalityMethod(final ReflectedMethod optionalityMethod) {
47          this.optionalityMethod = optionalityMethod;
48      }
49  
50      public void setShortNames(final List<String> shortNames) {
51          m_shortNames.addAll(shortNames);
52      }
53  
54      public void setLongName(final String longName) {
55          m_longName = longName;
56      }
57  
58      public void setDescription(final String description) {
59          m_description = description;
60      }
61  
62      public void setPattern(final String pattern) {
63          m_pattern = pattern;
64      }
65  
66      public void setDefaultValue(final List<String> defaultValue) {
67          m_defaultValue = defaultValue;
68      }
69  
70      public void setHelpRequest(final boolean helpRequest) {
71          m_helpRequest = helpRequest;
72      }
73  
74      public OptionSpecificationImpl createOptionSpecification() {
75          final OptionName optionName = new OptionName(m_longName, m_shortNames, m_description);
76          final OptionType optionType = new OptionType(m_type, m_pattern, m_multiValued);
77          final OptionContext optionContext = new OptionContext(m_defaultValue, m_helpRequest);
78  
79          return new OptionSpecificationImpl(optionName,
80                                              optionType,
81                                              optionContext,
82                                              m_method,
83                                              optionalityMethod);
84      }
85  }
