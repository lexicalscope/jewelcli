26  class InstanceArgumentPresentingStrategy<O> implements ArgumentPresentingStrategy<O> {
27      private final O options;
28      private final ReflectedClass<O> klass;
29      private final OptionsSpecification<O> specification;
30  
31      public InstanceArgumentPresentingStrategy(
32              final OptionsSpecification<O> specification,
33              final ReflectedClass<O> klass,
34              final O options) {
35          this.specification = specification;
36          this.klass = klass;
37          this.options = options;
38      }
39  
40      @Override public O presentArguments(final Map<String, Object> argumentMap) {
41          System.out.println(argumentMap);
42          for (final ReflectedMethod reflectedMethod : klass.methods(annotatedWith(Option.class))) {
43              final boolean isBoolean = specification.getSpecification(reflectedMethod).isBoolean();
44              setValueOnOptions(argumentMap, reflectedMethod, isBoolean);
45          }
46          for (final ReflectedMethod reflectedMethod : klass.methods(annotatedWith(Unparsed.class))) {
47              setValueOnOptions(argumentMap, reflectedMethod, false);
48          }
49          return options;
50      }
51  
52      private void setValueOnOptions(
53              final Map<String, Object> argumentMap,
54              final ReflectedMethod reflectedMethod,
55              final boolean isBoolean) {
56          if (argumentMap.containsKey(reflectedMethod.propertyName()))
57          {
58              if (isBoolean)
59              {
60                  reflectedMethod.call(options, argumentMap.containsKey(reflectedMethod.propertyName()));
61              }
62              else if (argumentMap.get(reflectedMethod.propertyName()) != null)
63              {
64                  reflectedMethod.call(options, argumentMap.get(reflectedMethod.propertyName()));
65              }
66          }
67      }
