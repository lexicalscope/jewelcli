28  public class ArgumentValidationExceptionMatcher {
29      public static Matcher<ArgumentValidationException> validationException(final ErrorType expectedErrorType) {
30          return new TypeSafeMatcher<ArgumentValidationException>() {
31              @Override public void describeTo(final Description description) {
32                  description
33                          .appendText(ArgumentValidationException.class.getSimpleName())
34                          .appendText(" with error type ")
35                          .appendValue(expectedErrorType);
36              }
37  
38              @Override protected boolean matchesSafely(final ArgumentValidationException item) {
39                  return selectFirst(item.getValidationErrors(), new TypeSafeMatcher<ValidationError>() {
40                      @Override public void describeTo(final Description description) {
41                          description
42                                  .appendText(ArgumentValidationException.ValidationError.class.getSimpleName())
43                                  .appendText(" with error type ")
44                                  .appendValue(expectedErrorType);
45                      }
46  
47                      @Override protected boolean matchesSafely(final ValidationError actualError) {
48                          return actualError.getErrorType().equals(expectedErrorType);
49                      }
50                  }) != null;
51              }
52          };
53      }
